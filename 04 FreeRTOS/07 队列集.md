//头文件
#include "queue_set.h"

// 定义队列句柄和队列集句柄
QueueHandle_t xQueue1, xQueue2;
QueueSetHandle_t xQueueSet;

typedef struct {uint8_t type;uint32_t data;TickType_t timestamp;} Message_t;

//创建队列集（容量=所有队列长度总和）

    initQueueSetDemo();

        // 根据xActivated分支处理不同队列数据
    }
}

## 案例

```c
#include "FreeRTOS.h"
#include "task.h"
#include "queue.h"
#include "queue_set.h"

// 定义队列句柄和队列集句柄
QueueHandle_t xQueue1, xQueue2;
QueueSetHandle_t xQueueSet;

// 定义消息结构
typedef struct {
    uint8_t type;
    uint32_t data;
    TickType_t timestamp;
} Message_t;

// 发送任务1 - 向队列1发送数据
void vSenderTask1(void *pvParameters) {
    Message_t message = {1, 0, 0};
    
    for(;;) {
        message.data++;
        message.timestamp = xTaskGetTickCount();
        
        // 发送消息到队列1
        if(xQueueSend(xQueue1, &message, portMAX_DELAY) == pdPASS) {
            printf("任务1发送: type=%d, data=%lu, time=%lu\n", 
                   message.type, message.data, message.timestamp);
        }
        
        vTaskDelay(pdMS_TO_TICKS(1000)); // 每1秒发送一次
    }
}

// 发送任务2 - 向队列2发送数据
void vSenderTask2(void *pvParameters) {
    Message_t message = {2, 1000, 0};
    
    for(;;) {
        message.data += 10;
        message.timestamp = xTaskGetTickCount();
        
        // 发送消息到队列2
        if(xQueueSend(xQueue2, &message, portMAX_DELAY) == pdPASS) {
            printf("任务2发送: type=%d, data=%lu, time=%lu\n", 
                   message.type, message.data, message.timestamp);
        }
        
        vTaskDelay(pdMS_TO_TICKS(500)); // 每0.5秒发送一次
    }
}

// 接收任务 - 从队列集中接收数据
void vReceiverTask(void *pvParameters) {
    QueueSetMemberHandle_t xActivatedMember;
    Message_t receivedMessage;
    BaseType_t xResult;
    
    printf("接收任务启动，等待队列集中的数据...\n");
    
    for(;;) {
        // 等待队列集中任意队列有数据可用（阻塞等待）
        xActivatedMember = xQueueSelectFromSet(xQueueSet, portMAX_DELAY);
        
        if(xActivatedMember != NULL) {
            // 判断是哪个队列被激活
            if(xActivatedMember == xQueue1) {
                // 从队列1读取数据
                xResult = xQueueReceive(xQueue1, &receivedMessage, 0);
                if(xResult == pdPASS) {
                    printf(">>> 从队列1收到: type=%d, data=%lu, time=%lu\n", 
                           receivedMessage.type, receivedMessage.data, receivedMessage.timestamp);
                }
            }
            else if(xActivatedMember == xQueue2) {
                // 从队列2读取数据
                xResult = xQueueReceive(xQueue2, &receivedMessage, 0);
                if(xResult == pdPASS) {
                    printf(">>> 从队列2收到: type=%d, data=%lu, time=%lu\n", 
                           receivedMessage.type, receivedMessage.data, receivedMessage.timestamp);
                }
            }
        }
    }
}

// 初始化函数
void initQueueSetDemo(void) {
    // 创建两个队列，每个队列可以存储5个Message_t结构
    xQueue1 = xQueueCreate(5, sizeof(Message_t));
    xQueue2 = xQueueCreate(5, sizeof(Message_t));
    
    // 创建队列集，容量为两个队列的总和
    xQueueSet = xQueueCreateSet(5 + 5);
    
    if(xQueue1 != NULL && xQueue2 != NULL && xQueueSet != NULL) {
        // 将队列添加到队列集中
        xQueueAddToSet(xQueue1, xQueueSet);
        xQueueAddToSet(xQueue2, xQueueSet);
        
        // 创建任务
        xTaskCreate(vSenderTask1, "发送者1", 1024, NULL, 2, NULL);
        xTaskCreate(vSenderTask2, "发送者2", 1024, NULL, 2, NULL);
        xTaskCreate(vReceiverTask, "接收者", 1024, NULL, 1, NULL);
        
        printf("队列集演示初始化完成\n");
    } else {
        printf("队列集创建失败!\n");
    }
}

int main(void) {
    // 硬件初始化...
    
    // 初始化队列集演示
    initQueueSetDemo();
    
    // 启动FreeRTOS调度器
    vTaskStartScheduler();
    
    for(;;);
}
```
