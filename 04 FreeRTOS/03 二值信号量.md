## 要点

```c
//头文件
#include "semphr.h"

//信号量句柄
SemaphoreHandle_t xBinarySemaphore = NULL;

// 创建二值信号量（适合「一次生产对应一次消费」）
// 初始状态：空（消费者无法获取，需生产者先释放）
//返回值：成功返回信号量句柄，失败返回NULL
xBinarySemaphore = xSemaphoreCreateBinary();

//获取信号量    返回值：pdTRUE/pdFALSE
xSemaphoreTake(xBinarySemaphore, portMAX_DELAY)

//释放信号量    返回值：pdTRUE/pdFALSE
xSemaphoreGive(xBinarySemaphore) 
```

## 案例
```c
#include "FreeRTOS.h"
#include "task.h"
#include "semphr.h"
#include "main.h"

// 定义任务句柄和信号量句柄
TaskHandle_t xProducerTaskHandle = NULL;
TaskHandle_t xConsumerTaskHandle = NULL;
SemaphoreHandle_t xBinarySemaphore = NULL;

// 生产者任务 - 释放信号量
void vProducerTask(void *pvParameters)
{
    TickType_t xLastWakeTime;
    const TickType_t xFrequency = 1000; // 1秒周期
    
    xLastWakeTime = xTaskGetTickCount();
    
    for(;;)
    {
        // 等待下一个周期
        vTaskDelayUntil(&xLastWakeTime, xFrequency);
        
        // 生产数据（模拟）
        printf("生产者: 数据准备完成\r\n");
        
        // 释放二值信号量，通知消费者
        if(xSemaphoreGive(xBinarySemaphore) == pdTRUE)
        {
            printf("生产者: 信号量已释放，通知消费者\r\n");
        }
        else
        {
            printf("生产者: 信号量释放失败（消费者尚未处理）\r\n");
        }
    }
}

// 消费者任务 - 获取信号量
void vConsumerTask(void *pvParameters)
{
    for(;;)
    {
        // 等待信号量（无限期等待）
        if(xSemaphoreTake(xBinarySemaphore, portMAX_DELAY) == pdTRUE)
        {
            // 成功获取信号量，处理数据
            printf("消费者: 收到信号量，开始处理数据...\r\n");
            
            // 模拟数据处理时间
            vTaskDelay(pdMS_TO_TICKS(500));
            
            printf("消费者: 数据处理完成\r\n");
        }
    }
}

// 初始化FreeRTOS组件
void vInitFreeRTOS(void)
{
    // 创建二值信号量
    xBinarySemaphore = xSemaphoreCreateBinary();
    
    if(xBinarySemaphore != NULL)
    {
        printf("二值信号量创建成功\r\n");
        
        // 创建生产者任务
        if(xTaskCreate(vProducerTask, "Producer", 128, NULL, 2, &xProducerTaskHandle) == pdPASS)
        {
            printf("生产者任务创建成功\r\n");
        }
        
        // 创建消费者任务
        if(xTaskCreate(vConsumerTask, "Consumer", 128, NULL, 1, &xConsumerTaskHandle) == pdPASS)
        {
            printf("消费者任务创建成功\r\n");
        }
        
        // 启动调度器
        vTaskStartScheduler();
    }
    else
    {
        printf("二值信号量创建失败\r\n");
    }
}

// 主函数
int main(void)
{
    // 硬件初始化
    HAL_Init();
    SystemClock_Config();
    
    // 初始化FreeRTOS
    vInitFreeRTOS();
    
    // 如果调度器启动失败，程序不会执行到这里
    for(;;);
}
```
