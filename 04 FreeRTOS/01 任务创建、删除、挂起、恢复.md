```c
vTaskDelete(Worker_TaskHandle);  // 根据句柄删除工作线程任务  
 Worker_TaskHandle = NULL;  // 任务删除后，句柄置为NULL  
      
```

```c

```

# 具体案例

这是一个基于 FreeRTOS 的多任务示例程序，包含 3 个核心任务：

1.  **LED_Task**：周期性翻转 LED 引脚（500ms 一次），并检测工作线程状态；
    
2.  **Worker_Task**：模拟 10 秒的工作流程，完成后自动删除自身；
    
3.  **Monitor_Task**：周期性（2 秒）监控系统状态（堆空间、任务状态），并自动重建已停止的工作线程；
    

同时提供了**外部删除工作线程**（按键触发）、**打印系统任务信息**等辅助功能，是 FreeRTOS 任务创建、删除、状态管理的典型演示代码。

#### 几个问题

> **1. 句柄是什么？有什么用？** 
> “句柄” 可以理解成资源的 “专属编号 / 身份证”—— 系统里的资源（比如文件、任务、设备）都有对应的内存数据，但我们不用直接操作这些数据的地址，而是通过 “句柄” 来间接指代这个资源。 作用：通过句柄能安全、快速地找到并操作对应的资源（比如操作文件时的 “文件句柄”，操作任务时的 “任务句柄”）。

> **2. 任务句柄（TaskHandle_t）是什么？有什么用？** 
> 在 FreeRTOS 里，任务句柄是每个任务的 “唯一标识”—— 本质是指向 “任务控制块（TCB）” 的指针（TCB 是 FreeRTOS 存储任务信息的结构体，包含任务的优先级、栈地址、运行状态等）。 核心作用：后续要对任务做操作（比如删除、修改优先级、查询状态、挂起 / 恢复）时，必须通过 “任务句柄” 告诉 FreeRTOS：“我要操作的是哪个任务”。 比如你代码里的Worker_TaskHandle： 创建任务时，xTaskCreate会把这个任务的句柄存到Worker_TaskHandle里； 后续调用vTaskDelete(Worker_TaskHandle)时，FreeRTOS 通过这个句柄找到对应的任务，执行删除操作。

> **3. 为什么初始设为NULL？** 
> 任务句柄初始设为NULL（空指针），是因为任务还没创建时，这个句柄对应的任务并不存在——NULL表示 “当前没有关联任何任务”。 后续可以通过检查句柄是否为NULL，判断任务是否已经创建（比如你代码里Delete_Worker_Task中先判断Worker_TaskHandle != NULL，就是确认任务存在后再执行删除）； 等调用xTaskCreate创建任务时，FreeRTOS 会自动把实际的任务句柄赋值给这个变量，覆盖初始的NULL。

/* 任务状态标志 */  
volatile uint8_t worker_task_running = 0;  // 工作线程运行状态标志（volatile防止编译器优化，多任务共享变量）  
​  
/* LED闪烁任务 */  
void LED_Task(void *argument)  //参数指针，用于创建任务时传递数据，argument为参数名可自定义

        }  
    }  
}

void Worker_Task(void *argument)  // 工作线程任务的入口函数  
{  
    uint32_t counter = 0;  // 定义计数变量，用于模拟任务执行进度  
    printf("工作线程已启动，将运行10秒后自动删除\r\n");  // 打印工作线程启动信息

{

    }

/* 外部删除任务函数 */  
void Delete_Worker_Task(void)  // 外部触发删除工作线程的函数  
{  
    if(Worker_TaskHandle != NULL && worker_task_running)  // 判断工作线程句柄非空且处于运行中  
    {

/* 获取任务信息 */  
void Print_Task_Info(void)  // 打印系统所有任务信息的函数  
{  
    TaskStatus_t *pxTaskStatusArray;  // 指向任务状态结构体数组的指针（存储任务信息）  
    volatile uint32_t uxArraySize;  // 任务数组的大小（即系统中任务数量）  
    uint32_t x;  // 循环变量  
    uint32_t ulTotalRunTime;  // 存储系统总运行时间（本代码暂未使用）  
      
    /* 获取任务数量 */  
    uxArraySize = uxTaskGetNumberOfTasks();  // 获取当前系统中的任务总数  
    printf("系统中共有 %d 个任务\r\n", uxArraySize);  // 打印任务数量  
      
    /* 分配内存存储任务状态 */  
    pxTaskStatusArray = pvPortMalloc(uxArraySize * sizeof(TaskStatus_t));  // 动态分配内存存储任务状态  
      
    if(pxTaskStatusArray != NULL)  // 判断内存分配是否成功  
    {  
        /* 获取任务状态信息 */  
        uxArraySize = uxTaskGetSystemState(pxTaskStatusArray,  // 存储任务状态的数组  
                                           uxArraySize,  // 数组最大容量  
                                           &ulTotalRunTime);  // 系统总运行时间的存储地址  
          
        for(x = 0; x < uxArraySize; x++)  // 遍历所有任务的状态信息  
        {  
            printf("任务：%s，状态：", pxTaskStatusArray[x].pcTaskName);  // 打印任务名称  
            switch(pxTaskStatusArray[x].eCurrentState)  // 根据任务状态枚举值判断状态  
            {  
                case eRunning:  
                    printf("运行中");  // 任务当前正在执行  
                    break;  
                case eReady:  
                    printf("就绪");  // 任务等待调度执行  
                    break;  
                case eBlocked:  
                    printf("阻塞");  // 任务因延时/等待事件处于阻塞态  
                    break;  
                case eSuspended:  
                    printf("挂起");  // 任务被主动挂起  
                    break;  
                case eDeleted:  
                    printf("已删除");  // 任务已删除（等待系统回收资源）  
                    break;  
                default:  
                    printf("未知");  // 未知状态  
                    break;  
            }  
            printf("，优先级：%lu\r\n", pxTaskStatusArray[x].uxCurrentPriority);  // 打印任务当前优先级  
        }  
          
        /* 释放内存 */  
        vPortFree(pxTaskStatusArray);  // 释放之前分配的任务状态数组内存  
    }
