## 要点
```c
//定义信号量句柄 
SemaphoreHandle_t xCountingSemaphore;

//创建计数型信号量              参数：最大计数值 初始值
xCountingSemaphore = xSemaphoreCreateCounting(10, 0);

//释放信号量   增加计数值           返回值：pdTRUE/pdFALSE
xSemaphoreGive(xCountingSemaphore)

//获取信号量（如果计数值为0则阻塞）  返回值：pdTRUE/pdFALSE
xSemaphoreTake(xCountingSemaphore, portMAX_DELAY)
```
###### 二值信号量，互斥信号量，计数型信号量的创建信号量的函数有区别，释放和获取信号量的函数没有区别
-----
## 案例
```c
#include "FreeRTOS.h"
#include "task.h"
#include "semphr.h"  //信号量头文件
#include "main.h"
#include "stdio.h"

/* 定义信号量句柄 */
SemaphoreHandle_t xCountingSemaphore;

/* 共享资源（模拟数据缓冲区） */
uint32_t dataBuffer[10];
uint8_t dataCount = 0;

/* 任务函数声明 */
void ProducerTask(void *pvParameters);
void ConsumerTask(void *pvParameters);

/* 初始化函数 */
void Semaphore_Init(void)
{
    /* 创建计数型信号量
     * 参数1：最大计数值
     * 参数2：初始计数值
     */
    xCountingSemaphore = xSemaphoreCreateCounting(10, 0);
    
    if (xCountingSemaphore == NULL) {
        printf("信号量创建失败！\r\n");
        Error_Handler();
    } else {
        printf("计数型信号量创建成功！\r\n");
    }
    
    /* 创建生产者任务 */
    if (xTaskCreate(ProducerTask, 
                   "Producer", 
                   configMINIMAL_STACK_SIZE, 
                   NULL, 
                   2,  // 优先级
                   NULL) != pdPASS) {
        printf("生产者任务创建失败！\r\n");
        Error_Handler();
    }
    
    /* 创建消费者任务 */
    if (xTaskCreate(ConsumerTask, 
                   "Consumer", 
                   configMINIMAL_STACK_SIZE, 
                   NULL, 
                   1,  // 优先级
                   NULL) != pdPASS) {
        printf("消费者任务创建失败！\r\n");
        Error_Handler();
    }
}

/* 生产者任务 - 产生数据 */
void ProducerTask(void *pvParameters)
{
    TickType_t xLastWakeTime;
    const TickType_t xFrequency = pdMS_TO_TICKS(1000); // 1秒周期
    
    xLastWakeTime = xTaskGetTickCount();
    
    while(1) {
        /* 等待下一个周期 */
        vTaskDelayUntil(&xLastWakeTime, xFrequency);
        
        /* 生产数据 */
        if (dataCount < 10) {
            dataBuffer[dataCount] = xTaskGetTickCount(); // 模拟数据
            dataCount++;
            printf("生产者：产生数据，当前数据量：%d\r\n", dataCount);
            
            /* 释放信号量 - 增加计数值 */
            if (xSemaphoreGive(xCountingSemaphore) == pdTRUE) {
                printf("生产者：释放信号量成功\r\n");
            } else {
                printf("生产者：信号量已达最大值！\r\n");
            }
        } else {
            printf("生产者：缓冲区已满！\r\n");
        }
        
        /* 打印信号量当前计数值 */
        UBaseType_t uxCount = uxSemaphoreGetCount(xCountingSemaphore);
        printf("当前信号量计数值：%ld\r\n", uxCount);
    }
}

/* 消费者任务 - 处理数据 */
void ConsumerTask(void *pvParameters)
{
    while(1) {
        /* 等待信号量 - 如果计数值为0则阻塞 */
        printf("消费者：等待数据...\r\n");
        
        if (xSemaphoreTake(xCountingSemaphore, portMAX_DELAY) == pdTRUE) {
            /* 成功获取信号量，处理数据 */
            if (dataCount > 0) {
                dataCount--;
                printf("消费者：处理数据，剩余数据量：%d\r\n", dataCount);
                
                /* 模拟数据处理时间 */
                vTaskDelay(pdMS_TO_TICKS(500));
            }
        }
    }
}

/* 在主函数中调用初始化 */
int main(void)
{
    HAL_Init();
    SystemClock_Config();
    
    /* 初始化信号量和任务 */
    Semaphore_Init();
    
    /* 启动FreeRTOS调度器 */
    vTaskStartScheduler();
    
    while(1) {
        /* 正常情况下不会执行到这里 */
    }
}
```
