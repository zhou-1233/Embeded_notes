# 要点
-----------
#### 获取单个任务状态
```c
/* 方法1: 使用eTaskGetState获取单个任务状态 */
        eTaskState ledState = eTaskGetState(LED_TaskHandle);  // 获取LED任务的当前状态（枚举类型）
        eTaskState workerState = eTaskGetState(Worker_TaskHandle);  // 获取工作任务的当前状态
        eTaskState statusState = eTaskGetState(Status_TaskHandle);  // 获取自身任务的当前状态
```

#### 获取所有任务状态
```c
/* 方法2: 使用uxTaskGetSystemState获取所有任务状态 */
UBaseType_t taskCount = uxTaskGetNumberOfTasks();  // 获取系统当前存在的任务总数
```

#### 头文件
```c
#include "FreeRTOS.h"          // 包含FreeRTOS核心功能头文件，提供基础类型和宏定义
#include "task.h"              // 包含FreeRTOS任务管理相关头文件，如任务创建、延时、状态查询等API

#include "usart.h"             // 包含串口外设驱动头文件，提供HAL_UART_Transmit等串口操作函数

#### 任务句柄声明
```c
// 任务句柄定义
TaskHandle_t LED_TaskHandle = NULL;          // LED任务句柄，用于后续操作该任务（如查询状态），初始化为NULL
TaskHandle_t Status_TaskHandle = NULL;       // 状态查询任务句柄，初始化为NULL
TaskHandle_t Worker_TaskHandle = NULL;       // 工作模拟任务句柄，初始化为NULL

// 任务状态查询结果结构体（本代码自定义，仅声明未实际使用，预留扩展用）
typedef struct {
    char taskName[20];                       // 存储任务名称的字符数组
    eTaskState taskState;                    // 存储任务状态（FreeRTOS内置枚举类型）
    uint32_t runTimeCounter;                 // 存储任务累计运行时间计数器值
    UBaseType_t stackHighWaterMark;          // 存储任务栈高水位线（最小剩余栈空间）
} TaskStatusInfo;
```

#### 具体任务的实现
```c

/* LED闪烁任务 */
void LED_Task(void *argument)                // LED任务执行函数，argument为任务入参（本任务未使用）
{
    while(1) {                               // FreeRTOS任务核心：无限循环，保证任务持续运行
        HAL_GPIO_TogglePin(LED_GPIO_Port, LED_Pin);  // 翻转LED引脚电平，实现LED闪烁
        
        // 模拟不同的任务状态（通过不同阻塞时间）
        static int counter = 0;              // 静态计数器，仅初始化一次，用于控制阻塞时间切换
        if(counter % 5 == 0) {               // 每5次循环触发短阻塞
            // 每5次循环阻塞100ms
            vTaskDelay(pdMS_TO_TICKS(100));  // 任务进入阻塞态100ms，pdMS_TO_TICKS将毫秒转为系统节拍数
        } else {                             // 其他次数触发长阻塞
            // 其他时间阻塞500ms
            vTaskDelay(pdMS_TO_TICKS(500));  // 任务进入阻塞态500ms
        }
        counter++;                           // 计数器自增，控制阻塞时间切换逻辑
    }
}

/* 工作模拟任务 */
void Worker_Task(void *argument)             // 工作模拟任务执行函数，argument为任务入参（本任务未使用）
{
    while(1) {                               // 任务无限循环
        // 模拟工作负载（消耗CPU时间）
        for(int i = 0; i < 1000000; i++) {   // 循环执行空操作，模拟CPU密集型任务
            __NOP(); // 空操作，消耗CPU时间  // 空指令，仅占用CPU周期，无实际运算/IO操作
        }
        
        // 进入阻塞状态，让出CPU资源
        vTaskDelay(pdMS_TO_TICKS(200));      // 任务阻塞200ms，期间CPU可调度其他任务
    }
}
```

#### 查询任务状态的函数实现
```c
/* 状态查询任务 */
void Status_Task(void *argument)             // 状态查询任务执行函数，argument为任务入参（本任务未使用）
{
    char statusBuffer[200];                  // 定义状态信息缓冲区，用于拼接串口输出的字符串
    
    while(1) {                               // 任务无限循环，持续查询并输出任务状态
        // 清空缓冲区，避免残留旧数据
        memset(statusBuffer, 0, sizeof(statusBuffer));  // 将缓冲区所有字节置0，初始化字符串
        
        /* 方法1: 使用eTaskGetState获取单个任务状态 */
        eTaskState ledState = eTaskGetState(LED_TaskHandle);  // 获取LED任务的当前状态
        eTaskState workerState = eTaskGetState(Worker_TaskHandle);  // 获取工作任务的当前状态
        eTaskState statusState = eTaskGetState(Status_TaskHandle);  // 获取自身任务的当前状态
        
        // 状态转换为可读字符串（局部函数，仅在Status_Task内生效）
        const char* getStateString(eTaskState state) {
            switch(state) {                   // 根据FreeRTOS任务状态枚举值分支
                case eRunning: return "Running";  // 运行态：任务当前正在执行
                case eReady: return "Ready";      // 就绪态：任务准备好执行，等待CPU
                case eBlocked: return "Blocked";  // 阻塞态：任务等待事件（如延时、信号量）
                case eSuspended: return "Suspended";  // 挂起态：任务被手动挂起
                case eDeleted: return "Deleted";  // 已删除态：任务被删除（未清理资源）
                case eInvalid: return "Invalid";  // 无效态：任务句柄无效
                default: return "Unknown";        // 未知状态：兼容未定义的枚举值
            }
        }
        
        /* 方法2: 使用uxTaskGetSystemState获取所有任务状态 */
        UBaseType_t taskCount = uxTaskGetNumberOfTasks();  // 获取系统当前存在的任务总数
        // 动态分配内存，存储所有任务的状态信息（按任务数×单个任务状态结构体大小）
        TaskStatus_t *taskStatusArray = pvPortMalloc(taskCount * sizeof(TaskStatus_t));
        
        if(taskStatusArray != NULL) {        // 检查内存分配是否成功，避免空指针操作
            // 获取系统所有任务的详细状态信息，返回实际获取的任务数
            taskCount = uxTaskGetSystemState(taskStatusArray, taskCount, NULL);
            
            // 构建状态信息字符串（拼接任务总数标题）
            char temp[50];                    // 临时缓冲区，用于拼接单行状态信息
            sprintf(statusBuffer, "=== 系统任务状态(共%d个任务) ===\r\n", taskCount);
            
            // 遍历所有任务，拼接每个任务的详细信息
            for(UBaseType_t i = 0; i < taskCount; i++) {
                // 获取该任务的栈高水位线（最小剩余栈空间，越小越接近栈溢出）
                UBaseType_t highWaterMark = uxTaskGetStackHighWaterMark(taskStatusArray[i].xHandle);
                
                sprintf(temp, "任务: %s\r\n", taskStatusArray[i].pcTaskName);  // 拼接任务名称
                strcat(statusBuffer, temp);   // 将临时字符串追加到主缓冲区
                // 拼接任务状态（调用局部函数转换为可读字符串）
                sprintf(temp, "  状态: %s\r\n", getStateString(taskStatusArray[i].eCurrentState));
                strcat(statusBuffer, temp);
                // 拼接任务当前优先级（强制转换为unsigned long避免格式警告）
                sprintf(temp, "  优先级: %lu\r\n", (unsigned long)taskStatusArray[i].uxCurrentPriority);
                strcat(statusBuffer, temp);
                // 拼接栈高水位线（单位：字，32位系统1字=4字节）
                sprintf(temp, "  栈高水位: %lu words\r\n", (unsigned long)highWaterMark);
                strcat(statusBuffer, temp);
                // 拼接任务累计运行时间计数器（系统节拍数）
                sprintf(temp, "  运行时间: %lu\r\n", (unsigned long)taskStatusArray[i].ulRunTimeCounter);
                strcat(statusBuffer, temp);
                
                strcat(statusBuffer, "----------------------------\r\n");  // 拼接分隔线，区分不同任务
            }
            
            vPortFree(taskStatusArray);       // 释放动态分配的内存，避免内存泄漏
        }
        
        // 通过串口2输出状态信息（阻塞式发送，超时设为最大）
        HAL_UART_Transmit(&huart2, (uint8_t*)statusBuffer, strlen(statusBuffer), HAL_MAX_DELAY);
        
        // 每2秒查询一次任务状态，控制输出频率
        vTaskDelay(pdMS_TO_TICKS(2000));      // 任务阻塞2秒，期间让出CPU
    }
}
```

#### 创建任务
```c
/* 任务创建函数 */
void Create_Tasks(void)                      // 封装任务创建逻辑，便于主函数调用
{
    // 创建LED任务
    xTaskCreate(LED_Task,           // 任务函数    // 指定任务执行的函数入口
                "LED_Task",         // 任务名称    // 任务名称（仅用于调试，最大长度由configMAX_TASK_NAME_LEN定义）
                128,               // 栈大小(words)  // 任务栈大小（单位：字，32位系统=128×4=512字节）
                NULL,              // 参数        // 传递给任务函数的入参（此处为NULL）
                2,                 // 优先级      // 任务优先级（数值越大优先级越高，范围0~configMAX_PRIORITIES-1）
                &LED_TaskHandle);  // 任务句柄    // 保存任务句柄的指针，用于后续操作任务
    
    // 创建工作任务
    xTaskCreate(Worker_Task,                // 指定工作任务的执行函数
                "Worker_Task",              // 工作任务名称
                256,                        // 工作任务栈大小（256字=1024字节）
                NULL,                       // 无入参传递
                3,                 // 较高优先级  // 优先级3，高于LED任务（优先获得CPU）
                &Worker_TaskHandle);        // 保存工作任务句柄
    
    // 创建状态查询任务
    xTaskCreate(Status_Task,                // 指定状态查询任务的执行函数
                "Status_Task",              // 状态查询任务名称
                512,               // 需要较大栈空间存储状态字符串  // 栈大小512字=2048字节（需存储拼接的字符串）
                NULL,                       // 无入参传递
                1,                 // 较低优先级  // 优先级1，低于其他任务（最后执行）
                &Status_TaskHandle);        // 保存状态查询任务句柄
}
```

#### FreeRTOS初始化
```c
/* 主函数 */
int main(void)                               // 程序入口函数
{
    HAL_Init();                              // 初始化STM32 HAL库（系统时钟、中断等）
    SystemClock_Config();                    // 配置系统时钟（由STM32CubeMX自动生成）
    MX_GPIO_Init();                          // 初始化GPIO外设（如LED引脚）
    MX_USART2_UART_Init();                   // 初始化USART2串口（用于输出任务状态）
    
    // 创建FreeRTOS任务
    Create_Tasks();                          // 调用任务创建函数，创建所有定义的任务
    
    // 启动调度器
    vTaskStartScheduler();                   // 启动FreeRTOS任务调度器，开始任务调度（从此处开始多任务执行）
    
    while(1) {                               // 死循环（正常情况下调度器启动后不会执行到此处）
        // 正常情况下不会执行到这里  // 仅在调度器启动失败（如内存不足）时执行
    }
}
