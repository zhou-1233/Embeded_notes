## 案例

 ```c
 #include "FreeRTOS.h"
#include "task.h"
#include "main.h"

// 任务句柄
TaskHandle_t xProducerTaskHandle = NULL;
TaskHandle_t xConsumerTaskHandle = NULL;

// 共享资源
static uint32_t ulSharedCounter = 0;

/*-----------------------------------------------------------*/

/**
 * @brief 生产者任务 - 发送通知（释放信号量）
 */
void vProducerTask(void *pvParameters)
{
    const TickType_t xDelay = pdMS_TO_TICKS(1000); // 1秒延迟
    uint32_t ulNotificationValue;
    
    for(;;)
    {
        // 模拟工作
        vTaskDelay(xDelay);
        
        // 临界区保护 - 修改共享资源
        taskENTER_CRITICAL();
        {
            ulSharedCounter++;
            printf("Producer: Counter = %lu\r\n", ulSharedCounter);
        }
        taskEXIT_CRITICAL();
        
        // 发送通知给消费者（相当于释放信号量）
        // 方法1：直接增加通知值（模拟计数信号量）
        xTaskNotifyGive(xConsumerTaskHandle);
        
        // 方法2：设置通知值并可选是否覆盖（更灵活）
        // xTaskNotify(xConsumerTaskHandle, 0, eIncrement);
        
        printf("Producer: Sent notification to consumer\r\n");
    }
}

/*-----------------------------------------------------------*/

/**
 * @brief 消费者任务 - 等待通知（获取信号量）
 */
void vConsumerTask(void *pvParameters)
{
    uint32_t ulNotificationCount;
    BaseType_t xResult;
    
    for(;;)
    {
        // 等待通知（相当于获取信号量）
        // ulTaskNotifyTake 参数说明：
        // pdTRUE: 收到通知后将通知值清零（模拟二值信号量）
        // pdFALSE: 收到通知后通知值减1（模拟计数信号量）
        // portMAX_DELAY: 无限期等待
        
        ulNotificationCount = ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
        
        if(ulNotificationCount > 0)
        {
            // 成功收到通知
            printf("Consumer: Received notification, count = %lu\r\n", ulNotificationCount);
            
            // 处理共享资源
            taskENTER_CRITICAL();
            {
                printf("Consumer: Processing counter = %lu\r\n", ulSharedCounter);
            }
            taskEXIT_CRITICAL();
        }
    }
}

/*-----------------------------------------------------------*/

/**
 * @brief 错误处理函数
 */
void vApplicationMallocFailedHook(void)
{
    printf("Memory allocation failed!\r\n");
    while(1);
}

/*-----------------------------------------------------------*/

/**
 * @brief 任务栈溢出钩子函数
 */
void vApplicationStackOverflowHook(TaskHandle_t xTask, char *pcTaskName)
{
    printf("Stack overflow in task: %s\r\n", pcTaskName);
    while(1);
}

/*-----------------------------------------------------------*/

/**
 * @brief 主函数 - 初始化并启动调度器
 */
int main(void)
{
    // 硬件初始化
    HAL_Init();
    SystemClock_Config();
    
    printf("FreeRTOS Task Notification Semaphore Demo Starting...\r\n");
    
    // 创建生产者任务
    xTaskCreate(
        vProducerTask,           // 任务函数
        "Producer",              // 任务名称
        256,                     // 栈大小
        NULL,                    // 参数
        tskIDLE_PRIORITY + 2,    // 优先级
        &xProducerTaskHandle     // 任务句柄
    );
    
    // 创建消费者任务
    xTaskCreate(
        vConsumerTask,           // 任务函数
        "Consumer",              // 任务名称
        256,                     // 栈大小
        NULL,                    // 参数
        tskIDLE_PRIORITY + 1,    // 优先级
        &xConsumerTaskHandle     // 任务句柄
    );
    
    // 启动FreeRTOS调度器
    vTaskStartScheduler();
    
    // 正常情况下不会执行到这里
    while(1);
}

/*-----------------------------------------------------------*/

/**
 * @brief 空闲任务钩子函数（可选）
 */
void vApplicationIdleHook(void)
{
    // 可以在这里执行低优先级任务
}
```
