## 要点
```c
//头文件
#include "event_groups.h"

/* 事件组句柄 */
EventGroupHandle_t xEventGroup;

/* 事件标志位定义 */
#define TASK1_READY_BIT     (1 << 0)  /* 位0: 任务1准备就绪 */
#define TASK2_COMPLETE_BIT  (1 << 1)  /* 位1: 任务2完成 */
#define ALL_EVENTS_BITS     (TASK1_READY_BIT | TASK2_COMPLETE_BIT)

/* 创建事件组 */ 
xEventGroup = xEventGroupCreate();

//设置事件位         事件组句柄    设置的事件位
xEventGroupSetBits(xEventGroup, TASK1_READY_BIT);

//清除事件位              句柄        事件位
xEventGroupClearBits(xEventGroup, TASK1_READY_BIT);

//读取事件位                                         句柄
EventBits_t uxCurrentBits = xEventGroupGetBits(xEventGroup);

//等待事件发生
uxBits = xEventGroupWaitBits(
    xEventGroup,        /* 事件组句柄 */
    ALL_EVENTS_BITS,    /* 等待的事件位 */
    pdFALSE,            /* 退出时是否自动清除事件位 */
    pdFALSE,            /* 是否等待所有事件位都发生 */
    xTicksToWait        /* 超时时间 */
);

// 删除事件组                句柄
void vEventGroupDelete(EventGroupHandle_t xEventGroup);
```

## 案例
```c
#include "FreeRTOS.h"
#include "task.h"
#include "event_groups.h"

/* 事件组句柄 */
EventGroupHandle_t xEventGroup;

/* 事件标志位定义 */
#define TASK1_READY_BIT     (1 << 0)  /* 位0: 任务1准备就绪 */
#define TASK2_COMPLETE_BIT  (1 << 1)  /* 位1: 任务2完成 */
#define ALL_EVENTS_BITS     (TASK1_READY_BIT | TASK2_COMPLETE_BIT)

/* 任务函数声明 */
void vTask1(void *pvParameters);
void vTask2(void *pvParameters);
void vEventMonitorTask(void *pvParameters);

int main(void)
{
    /* HAL库初始化 */
    HAL_Init();
    
    /* 创建事件组 */
    xEventGroup = xEventGroupCreate();
    
    if(xEventGroup != NULL)
    {
        /* 创建任务 */
        xTaskCreate(vTask1, "Task1", 128, NULL, 2, NULL);
        xTaskCreate(vTask2, "Task2", 128, NULL, 2, NULL);
        xTaskCreate(vEventMonitorTask, "EventMonitor", 128, NULL, 1, NULL);
        
        /* 启动FreeRTOS调度器 */
        vTaskStartScheduler();
    }
    
    /* 如果调度器启动失败，程序将执行到这里 */
    while(1);
}

/* 任务1 - 设置事件 */
void vTask1(void *pvParameters)
{
    TickType_t xLastWakeTime;
    const TickType_t xFrequency = 1000; /* 1秒周期 */
    
    xLastWakeTime = xTaskGetTickCount();
    
    for(;;)
    {
        /* 等待下一个周期 */
        vTaskDelayUntil(&xLastWakeTime, xFrequency);
        
        /* 模拟一些处理工作 */
        HAL_GPIO_TogglePin(GPIOA, GPIO_PIN_5); /* 翻转LED */
        
        /* 设置任务1就绪事件 */
        xEventGroupSetBits(xEventGroup, TASK1_READY_BIT);
        
        printf("Task1: 设置就绪事件\r\n");
    }
}

/* 任务2 - 设置完成事件 */
void vTask2(void *pvParameters)
{
    TickType_t xLastWakeTime;
    const TickType_t xFrequency = 2000; /* 2秒周期 */
    
    xLastWakeTime = xTaskGetTickCount();
    
    for(;;)
    {
        vTaskDelayUntil(&xLastWakeTime, xFrequency);
        
        /* 模拟处理工作 */
        printf("Task2: 执行处理...\r\n");
        
        /* 设置任务2完成事件 */
        xEventGroupSetBits(xEventGroup, TASK2_COMPLETE_BIT);
        
        printf("Task2: 设置完成事件\r\n");
    }
}

/* 事件监控任务 - 等待事件发生 */
void vEventMonitorTask(void *pvParameters)
{
    EventBits_t uxBits;
    const TickType_t xTicksToWait = pdMS_TO_TICKS(5000); /* 5秒超时 */
    
    for(;;)
    {
        /* 等待任意一个事件发生，不自动清除事件位 */
        uxBits = xEventGroupWaitBits(
            xEventGroup,        /* 事件组句柄 */
            ALL_EVENTS_BITS,    /* 等待的事件位 */
            pdFALSE,            /* 不自动清除事件位 */
            pdFALSE,            /* 不等待所有事件位 */
            xTicksToWait        /* 超时时间 */
        );
        
        if((uxBits & TASK1_READY_BIT) != 0)
        {
            printf("Monitor: 检测到任务1就绪事件\r\n");
            /* 手动清除事件位 */
            xEventGroupClearBits(xEventGroup, TASK1_READY_BIT);
        }
        
        if((uxBits & TASK2_COMPLETE_BIT) != 0)
        {
            printf("Monitor: 检测到任务2完成事件\r\n");
            xEventGroupClearBits(xEventGroup, TASK2_COMPLETE_BIT);
        }
        
        if((uxBits & ALL_EVENTS_BITS) == 0)
        {
            printf("Monitor: 等待事件超时\r\n");
        }
        
        /* 检查是否两个事件都发生了 */
        uxBits = xEventGroupGetBits(xEventGroup);
        if((uxBits & ALL_EVENTS_BITS) == ALL_EVENTS_BITS)
        {
            printf("Monitor: 所有事件都已发生！\r\n");
            /* 清除所有事件位 */
            xEventGroupClearBits(xEventGroup, ALL_EVENTS_BITS);
        }
    }
}
```
