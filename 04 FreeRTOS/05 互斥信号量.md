## 要点
```c
//创建互斥信号量句柄
SemaphoreHandle_t xMutex;

//创建互斥信号量        返回值：成功返回信号量句柄，失败返回NULL
xMutex = xSemaphoreCreateMutex();

//释放互斥信号量         返回值：pdTRUE/pdFALSE
xSemaphoreGive(xMutex);

//获取互斥信号量         返回值：pdTRUE/pdFALSE
xSemaphoreTake(xMutex, pdMS_TO_TICKS(500);
```

## 案例
```c
#include "FreeRTOS.h"
#include "task.h"
#include "semphr.h"
#include "main.h"
#include "usart.h"

/* 定义共享资源和互斥信号量 */
static int shared_counter = 0;
SemaphoreHandle_t xMutex;

/* 任务1函数 */
void Task1(void *pvParameters)
{
    TickType_t xLastWakeTime;
    const TickType_t xFrequency = 1000; // 1秒周期
    
    xLastWakeTime = xTaskGetTickCount();
    
    for(;;)
    {
        /* 等待获取互斥信号量 */
        if(xSemaphoreTake(xMutex, portMAX_DELAY) == pdTRUE)
        {
            /* 临界区开始 - 安全访问共享资源 */
            shared_counter++;
            printf("Task1: Counter = %d\r\n", shared_counter);
            
            /* 释放互斥信号量 */
            xSemaphoreGive(xMutex);
            /* 临界区结束 */
        }
        
        vTaskDelayUntil(&xLastWakeTime, xFrequency);
    }
}

/* 任务2函数 */
void Task2(void *pvParameters)
{
    TickType_t xLastWakeTime;
    const TickType_t xFrequency = 1500; // 1.5秒周期
    
    xLastWakeTime = xTaskGetTickCount();
    
    for(;;)
    {
        /* 尝试获取互斥信号量，最多等待500ms */
        if(xSemaphoreTake(xMutex, pdMS_TO_TICKS(500)) == pdTRUE)
        {
            /* 临界区开始 - 安全访问共享资源 */
            shared_counter += 2;
            printf("Task2: Counter = %d\r\n", shared_counter);
            
            /* 释放互斥信号量 */
            xSemaphoreGive(xMutex);
            /* 临界区结束 */
        }
        else
        {
            printf("Task2: 无法获取互斥信号量，资源被占用\r\n");
        }
        
        vTaskDelayUntil(&xLastWakeTime, xFrequency);
    }
}

/* 初始化函数 */
void MX_FREERTOS_Init(void)
{
    /* 创建互斥信号量 */
    xMutex = xSemaphoreCreateMutex();
    
    if(xMutex == NULL)
    {
        printf("互斥信号量创建失败！\r\n");
        return;
    }
    
    /* 创建任务 */
    xTaskCreate(Task1, "Task1", 128, NULL, 2, NULL);
    xTaskCreate(Task2, "Task2", 128, NULL, 2, NULL);
    
    printf("FreeRTOS 初始化完成\r\n");
}
```
