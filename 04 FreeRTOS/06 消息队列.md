## 要点
```c
 //定义消息结构体
typedef struct { uint8_t sensor_id;uint16_t sensor_value;TickType_t timestamp;} SensorData_t;

 //队列句柄
QueueHandle_t xSensorQueue;

//创建消息队列 - 可以存储10个SensorData_t结构体          返回值；队列句柄/NULL
xSensorQueue = xQueueCreate(10, sizeof(SensorData_t));

//发送数据到队列（入队）  队列句柄    要发送的数据  等待时间（100ms）  返回值；pdPASS/errQUEUE_FULL
xStatus = xQueueSend(xSensorQueue,&xSensorData, pdMS_TO_TICKS(100))     

// 从队列接收数据（出队)    队列句柄     接收缓存区      无限等待     返回值；pdPASS/pdFALSE
xStatus = xQueueReceive(xSensorQueue,&xReceivedData,portMAX_DELAY);

/* 获取队列中当前项目数量 */
UBaseType_t uxMessagesWaiting = uxQueueMessagesWaiting(xSensorQueue);

/* 获取队列剩余空间 */
UBaseType_t uxSpacesAvailable = uxQueueSpacesAvailable(xSensorQueue);

// 重置队列（清空所有消息）       返回值；pdPASS/pdFALSE
BaseType_t xQueueReset(QueueHandle_t);

//删除队列
void vQueueDelete(QueueHandle_t xQueue);
```

## 案例
```c
#include "FreeRTOS.h"
#include "task.h"
#include "queue.h"
#include "main.h"
#include "usart.h"

/* 定义消息结构体 */
typedef struct {
    uint8_t sensor_id;
    uint16_t sensor_value;
    TickType_t timestamp;
} SensorData_t;

/* 队列句柄 */
QueueHandle_t xSensorQueue;

/* 任务函数声明 */
void vSensorTask(void *pvParameters);
void vProcessingTask(void *pvParameters);

int main(void)
{
    HAL_Init();
    SystemClock_Config();
    MX_USART2_UART_Init();  // 用于调试输出
    
    /* 创建消息队列 - 可以存储10个SensorData_t结构体 */
    xSensorQueue = xQueueCreate(10, sizeof(SensorData_t));
    
    if (xSensorQueue == NULL) {
        printf("队列创建失败!\r\n");
        while(1);
    } else {
        printf("队列创建成功!\r\n");
    }
    
    /* 创建生产者任务（传感器任务） */
    xTaskCreate(vSensorTask,     // 任务函数
                "SensorTask",    // 任务名称
                128,            // 堆栈大小
                NULL,           // 参数
                2,              // 优先级
                NULL);          // 任务句柄
    
    /* 创建消费者任务（处理任务） */
    xTaskCreate(vProcessingTask, 
                "ProcessTask", 
                128, 
                NULL, 
                1,              // 优先级低于传感器任务
                NULL);
    
    /* 启动调度器 */
    vTaskStartScheduler();
    
    while (1) {}
}

/* 传感器任务 - 生产者 */
void vSensorTask(void *pvParameters)
{
    SensorData_t xSensorData;
    BaseType_t xStatus;
    uint16_t counter = 0;
    
    while(1) {
        /* 模拟传感器数据 */
        xSensorData.sensor_id = 1;
        xSensorData.sensor_value = counter++;
        xSensorData.timestamp = xTaskGetTickCount();
        
        /* 发送数据到队列（入队） */
        xStatus = xQueueSend(xSensorQueue,     // 队列句柄
                            &xSensorData,      // 要发送的数据
                            pdMS_TO_TICKS(100) // 等待时间(100ms)
                            );
        
        if (xStatus == pdPASS) {
            printf("数据发送成功: ID=%d, Value=%d\r\n", 
                   xSensorData.sensor_id, xSensorData.sensor_value);
        } else {
            printf("队列已满，数据发送失败!\r\n");
        }
        
        vTaskDelay(pdMS_TO_TICKS(500));  // 每500ms发送一次
    }
}

/* 处理任务 - 消费者 */
void vProcessingTask(void *pvParameters)
{
    SensorData_t xReceivedData;
    BaseType_t xStatus;
    
    while(1) {
        /* 从队列接收数据（出队） */
        xStatus = xQueueReceive(xSensorQueue,      // 队列句柄
                               &xReceivedData,     // 接收缓冲区
                               portMAX_DELAY       // 无限等待
                               );
        
        if (xStatus == pdPASS) {
            /* 计算数据延迟 */
            TickType_t current_time = xTaskGetTickCount();
            TickType_t delay = current_time - xReceivedData.timestamp;
            
            printf("数据处理: ID=%d, Value=%d, 延迟=%ldms\r\n", 
                   xReceivedData.sensor_id, 
                   xReceivedData.sensor_value,
                   delay * portTICK_PERIOD_MS);
            
            /* 这里可以添加实际的数据处理代码 */
        }
    }
}
```
