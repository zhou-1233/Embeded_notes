### 串口通信
```c
//轮询模式（CPU一直被占用）
//串口发送         句柄     要发送的数据      发送数据的字节长度   超时时间
HAL_UART_Transmit(&huart2,(uint8_t*)message, strlen(message),1000);
//串口接收
HAL_UART_Receive(&huart2,rxBuffer,10,1000);

//中断模式(更节约CPU资源)
HAL_UART_Transmit_IT(&huart2,(uint8_t*)message,strlen(message));
HAL_UART_Receive_IT(&huart2,rxBuffer,10);
//中断回调函数
HAL_UART_RxCpltCallback(*huart){
	//接收数据之后的处理逻辑
}

//DMA模式（最节约CPU资源）
HAL_UART_Transmit_DMA(&huart2,(uint8_t*)message,strlen(message));
AL_UART_Receive_DMA(&huart2,(uint8_t*)message,10);

```
### 定时器
```C
// 启动基本定时器 返回值：HAL_OK: 发送成功，HAL_ERROR:发送失败
HAL_TIM_Base_Start(&htim2);
//开启中断模式后，会有相应的中断回调函数，在到达一定时间（由预分频器和自动重装载寄存器）后会触发该中断回调函数

// 停止定时器
HAL_TIM_Base_Stop(&htim2);

// 获取计数器的值
__HAL_TIM_GET_COUNTER(&htim2);

// 设置计数器的值
__HAL_TIM_SET_COUNTER(&htim2, 0);

// 获取自动重装载寄存器的值
__HAL_TIM_GET_AUTORELOAD(&htim2);

// 设置自动重装载寄存器的值
__HAL_TIM_SET_AUTORELOAD(&htim2,100);

// 设置预分频器的值
__HAL_TIM_SET_PRESCALER(&htim2, 71);

// 开启输入捕获
HAL_TIM_IC_Start(&htim,channel);

// 开启输入捕获中断
HAL_TIM_IC_Start_IT(&htim,channel);// 设置比较寄存器的值

//设置比较寄存器的值
HAL_TIM_SET_COMPARE(&htim,channel,value);

// 获取比较寄存器的值
HAL_TIM_SET_COMPARE(&htim,channel);

// 清除CC1标志位
HAL_TIM_CLEAR_FLAG(&htim,TIM,CHANNEL);

// 启动PWM输出 句柄 输出通道
HAL_TIM_PWM_Start(&htim3, TIM_CHANNEL_1);

```

### 中断
```c

// 外部中断回调函数 触发中断的引脚编号   在图形化界面配置对应引脚为中断模式，并选择中断触发方式（上升沿或下降沿） 
HAL_GPIO_EXTI_Callback(GPIO_PIN_13)
{
	//要实现的逻辑
}

// 串口接收完毕回调函数 触发中断的串口句柄
HAL_UART_RxCpltCallback(&huart)
{
	//串口接收完毕后触发中断，执行该函数（处理接收到的数据）
}
//回调函数是在接收或发送完成后才会回调的函数，所以可以确定什么时候发送或接收完成。让程序在完成发送或接收后自动触发后续逻辑。
//相当于发送或接收完成后通知，确定什么时候发送完成

```
