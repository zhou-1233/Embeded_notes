### 01.LED

```C
// 定义LED总数（8个）
#define LED_NUM 8

/**
  * @brief  单个LED点亮
  * @param  led_num: LED编号(1~8)
  * @retval 无
  */
void LED_On(uint8_t led_num)
{
    // 校验LED编号范围
    if(led_num < 1 || led_num > LED_NUM) return;
    // 实际引脚：GPIOC的8~15引脚，LED1对应PC8，LED8对应PC15
    HAL_GPIO_WritePin(GPIOC, GPIO_PIN_8 << (led_num - 1), GPIO_PIN_RESET);
}

/**
  * @brief  单个LED熄灭
  * @param  led_num: LED编号(1~8)
  * @retval 无
  */
void LED_Off(uint8_t led_num)
{
    if(led_num < 1 || led_num > LED_NUM) return;
    HAL_GPIO_WritePin(GPIOC, GPIO_PIN_8 << (led_num - 1), GPIO_PIN_SET);
}

/**
  * @brief  单个LED状态翻转
  * @param  led_num: LED编号(1~8)
  * @retval 无
  */
void LED_Toggle(uint8_t led_num)
{
    if(led_num < 1 || led_num > LED_NUM) return;
    HAL_GPIO_TogglePin(GPIOC, GPIO_PIN_8 << (led_num - 1));
}

/**
  * @brief  按8位值控制全部LED
  * @param  led_val: 8位控制值(bit0~7对应LED1~8, 0亮1灭)
  * @retval 无
  */
void LED_Write(uint8_t led_val)
{
    uint16_t pin_state = 0x0000;
    // 按位映射: bit0→PC8, bit1→PC9 ... bit7→PC15
    for(uint8_t i=0; i<LED_NUM; i++)
    {
        if( !(led_val & (1 << i)) ) // 对应位为0则亮
            pin_state |= (GPIO_PIN_8 << i);
    }
    // 先置位所有LED引脚（灭）,再复位需要点亮的引脚（亮）
    HAL_GPIO_WritePin(GPIOC, GPIO_PIN_8 | GPIO_PIN_9 | GPIO_PIN_10 | GPIO_PIN_11 |
                             GPIO_PIN_12 | GPIO_PIN_13 | GPIO_PIN_14 | GPIO_PIN_15, GPIO_PIN_SET);
    HAL_GPIO_WritePin(GPIOC, pin_state, GPIO_PIN_RESET);
}

/**
  * @brief  LED流水灯(左→右→左循环)
  * @param  delay_ms: 单灯延时(ms)
  * @retval 无
  */
void LED_Flow(uint16_t delay_ms)
{
    static uint8_t dir = 0;  // 0:左→右 1:右→左
    static uint8_t cnt = 1;

    LED_Off( (cnt == 1) ? 8 : (cnt - 1) ); // 前一个灭
    LED_On(cnt);                          // 当前亮
    HAL_Delay(delay_ms);

    // 方向控制
    if(dir == 0) cnt++;
    else cnt--;

    if(cnt >= LED_NUM) dir = 1;
    if(cnt <= 1) dir = 0;
}
```

### 02.按键

```c 
#define KEY_NUM 4                // 按键总数（蓝桥杯开发板默认4个按键）
// 按键状态枚举定义（需在代码开头声明，避免编译报错）
typedef enum {
    KEY_RELEASE = 0,  // 按键释放
    KEY_DOWN,         // 按键按下中
    KEY_CLICK,        // 按键单击（按下后松开）
    KEY_LONG          // 按键长按
} KeyState;

// ========================== 内部状态变量 ==========================
static uint8_t      key_scan_buf[KEY_NUM] = {0};    // 消抖缓冲（每个按键8位，用于存储连续8次扫描值）
static uint16_t     key_press_tick[KEY_NUM] = {0};  // 按下计时（ms），用于判断长按
static KeyState     key_state[KEY_NUM] = {0};       // 每个按键的最终状态
#define KEY_LONG_TICK    800   // 长按阈值(ms)：按下超过800ms判定为长按
#define KEY_DEBOUNCE_TICK 10   // 消抖时间(ms)：连续10ms电平稳定才确认状态

/**
  * @brief  按键初始化
  * @note   1. 开启TIM2定时器1ms中断（用于按键扫描）
  *         2. 需确保TIM2已配置为1ms中断（ARR=7199, PSC=9，时钟72MHz）
  */
void KEY_Init(void)
{
    // 开启TIM2定时器中断（1ms触发一次，用于按键扫描和消抖）
    HAL_TIM_Base_Start_IT(&htim2); 
}

/**
  * @brief  TIM2 1ms 中断回调函数 → 核心按键扫描+消抖逻辑
  * @param  htim: 定时器句柄
  * @note   1. 按键硬件：KEY1=PA0, KEY2=PA1, KEY3=PA2, KEY4=PA3
  *         2. 按键电平：按下为低电平（0），释放为高电平（1）
  */
void HAL_TIM_PeriodElapsedCallback(TIM_HandleTypeDef *htim)
{
    // 仅处理TIM2的1ms中断
    if(htim->Instance == TIM2)
    {
        // 遍历4个按键，逐个扫描
        for(uint8_t i=0; i<KEY_NUM; i++)
        {
            uint8_t now = 0;  // 当前按键电平（0=按下，1=释放）
            // 根据按键序号读取对应引脚电平（无别名，直接用实际引脚）
            switch(i)
            {
                case 0: now = !HAL_GPIO_ReadPin(GPIOA, GPIO_PIN_0); break;  // KEY1: PA0
                case 1: now = !HAL_GPIO_ReadPin(GPIOA, GPIO_PIN_1); break;  // KEY2: PA1
                case 2: now = !HAL_GPIO_ReadPin(GPIOA, GPIO_PIN_2); break;  // KEY3: PA2
                case 3: now = !HAL_GPIO_ReadPin(GPIOA, GPIO_PIN_3); break;  // KEY4: PA3
                default: break;
            }
            
            // 消抖滤波：8位移位寄存器，每次左移1位，最低位存入当前电平
            // 连续8次扫描值都相同 → 电平稳定
            key_scan_buf[i] = (key_scan_buf[i] << 1) | now;
            
            // 连续10ms稳定（1ms中断，需连续10次扫描值相同 → key_scan_buf[i]为0x00或0xFF）
            if(key_scan_buf[i] == 0x00)       // 稳定松开：连续10次扫描都是释放状态（0）
            {
                if(key_state[i] == KEY_DOWN)  // 之前是按下中 → 现在松开，判定为单击
                {
                    key_state[i] = KEY_CLICK;
                }
                key_press_tick[i] = 0;        // 按下计时清零
            }
            else if(key_scan_buf[i] == 0xFF)  // 稳定按下：连续10次扫描都是按下状态（1）
            {
                // 按下计时累加（未到长按阈值则继续计时）
                if(key_press_tick[i] < KEY_LONG_TICK)
                {
                    key_press_tick[i]++;
                }
                // 判断是否达到长按阈值
                if(key_press_tick[i] >= KEY_LONG_TICK)
                {
                    key_state[i] = KEY_LONG;   // 按下超过800ms → 判定为长按
                }
                else
                {
                    key_state[i] = KEY_DOWN;   // 按下中（未到长按阈值）
                }
            }
        }
    }
}

/**
  * @brief  按键状态处理（需在主循环中调用）
  * @note   单击状态（KEY_CLICK）仅保持1次扫描，防止重复触发
  *         例：主循环中检测到KEY_CLICK后，立即将状态置为RELEASE
  */
void KEY_ScanProcess(void)
{
    // 遍历所有按键，处理单击状态的清零
    for(uint8_t i=0; i<KEY_NUM; i++)
    {
        if(key_state[i] == KEY_CLICK)
        {
            key_state[i] = KEY_RELEASE;  // 单击状态仅触发一次
        }
    }
}

/**
  * @brief  获取指定按键的状态
  * @param  key: 按键编号（1~4，对应KEY1~KEY4）
  * @retval KeyState: 按键状态（RELEASE/DOWN/CLICK/LONG）
  * @note   支持多键同时判断，无别名直接映射按键编号和引脚
  */
KeyState KEY_GetState(uint8_t key)
{
    // 校验按键编号范围（1~4），非法编号返回释放状态
    if(key < 1 || key > KEY_NUM) return KEY_RELEASE;
    // 按键编号1→数组索引0（KEY1），以此类推
    return key_state[key - 1];
}
```

```c
  * @brief  连续模式下读取当前值
  * @retval ADC原始值
  */

    {
        return HAL_ADC_GetValue(&ADC_HANDLE);
    }
    return 0;

/**
  * @brief  软件平均滤波（多次采样求和取平均）
  * @param  times：采样次数
  * @retval 滤波后ADC值
  */
uint16_t ADC_GetAverageSample(uint8_t times)
{
    uint32_t sum = 0;
    // 边界保护
    if(times == 0) times = 1;
    if(times > 100) times = 100;

    // 多次采样
    for(uint8_t i=0; i<times; i++)
    {
        sum += ADC_GetSingleSample();
        HAL_Delay(1); // 间隔采样，减小干扰
    }

    // 返回平均值
    return (uint16_t)(sum / times);
}
```

### 06.输入捕获

```c
// 捕获数据
volatile uint32_t IC_Freq = 0;
volatile uint32_t IC_Duty = 0;
volatile uint8_t  IC_ValidFlag = 0;

// 内部变量
static uint32_t IC_Period = 0;  // 周期(us)
static uint8_t  IC_Stage = 0;   // 0:等待上升沿 1:等待下降沿

/**
  * @brief  输入捕获初始化（开启中断+捕获）
  */
void IC_Init(void)
{
    // 先启动上升沿捕获
    HAL_TIM_IC_Start_IT(&IC_HTIM, IC_CHANNEL);
}

/**
  * @brief  输入捕获中断回调函数
  */
void HAL_TIM_IC_CaptureCallback(TIM_HandleTypeDef *htim)
{
    if(htim->Instance == TIM3)
    {
        if(htim->Channel == HAL_TIM_ACTIVE_CHANNEL_1)
        {
            if(IC_Stage == 0) // 第一步：捕获上升沿 → 读周期
            {
                // 读取周期计数值（1us=1）
                IC_Period = HAL_TIM_ReadCapturedValue(&IC_HTIM, IC_CHANNEL);
                
                // 切换为下降沿捕获
                __HAL_TIM_SET_CAPTUREPOLARITY(&IC_HTIM, IC_CHANNEL, 
                    TIM_INPUTCHANNELPOLARITY_FALLING);
                IC_Stage = 1;
            }
            else if(IC_Stage == 1) // 第二步：捕获下降沿 → 读脉宽
            {
                // 读取高电平计数值
                IC_Duty = HAL_TIM_ReadCapturedValue(&IC_HTIM, IC_CHANNEL);
                
                // 切回上升沿，等待下一个周期
                __HAL_TIM_SET_CAPTUREPOLARITY(&IC_HTIM, IC_CHANNEL, 
                    TIM_INPUTCHANNELPOLARITY_RISING);
                IC_Stage = 0;
                
                // 计算频率（防除0）
                if(IC_Period != 0)
                {
                    IC_Freq = 1000000 / IC_Period;
                }
                else
                {
                    IC_Freq = 0;
                    IC_Duty = 0;
                }
                IC_ValidFlag = 1; // 捕获完成
            }
        }
    }
}
```

```c
/**
  * @brief  PWM初始化（开启PWM输出）
  */
void PWM_Init(void)
{
    // 启动PWM通道
    HAL_TIM_PWM_Start(&PWM_HANDLE, PWM_CHANNEL);
    // 初始占空比0
    __HAL_TIM_SET_COMPARE(&PWM_HANDLE, PWM_CHANNEL, 0);
}

/**
  * @brief  设置PWM占空比
  * @param  duty: 0~100 对应 0%~100%
  */
void PWM_SetDuty(uint8_t duty)
{
    if(duty > 100) duty = 100;
    uint32_t ccr = (duty * PWM_PERIOD) / 100;
    __HAL_TIM_SET_COMPARE(&PWM_HANDLE, PWM_CHANNEL, ccr);
}

/**
  * @brief  呼吸灯（渐变亮→渐变灭）
  * @param  delay_ms: 渐变步长延时(越小越快)
  */
void PWM_Breath(uint16_t delay_ms)
{
    static int8_t dir = 1;   // 1=变亮, -1=变暗
    static uint8_t duty = 0;

    duty += dir;

    // 边界换向
    if(duty >= 100) dir = -1;
    if(duty <= 0)   dir = 1;

    PWM_SetDuty(duty);
    HAL_Delay(delay_ms);
}
```

```c
{
    HAL_I2C_Mem_Write(&EEPROM_I2C, EEPROM_ADDR_W, addr, I2C_MEMADD_SIZE_8BIT, &data, 1, 100);
    HAL_Delay(5); // 写入等待周期
}

/**
  * @brief  连续读多个字节
  */
void EEPROM_ReadBuffer(uint16_t addr, uint8_t *pbuf, uint16_t len)
{
    HAL_I2C_Mem_Read(&EEPROM_I2C, EEPROM_ADDR_R, addr, I2C_MEMADD_SIZE_8BIT, pbuf, len, 100);
}

/**
  * @brief  连续写（自动分页）
  */
void EEPROM_WriteBuffer(uint16_t addr, uint8_t *pbuf, uint16_t len)
{
    uint16_t page_offset = addr % EEPROM_PAGE_SIZE;
    uint16_t write_size = EEPROM_PAGE_SIZE - page_offset;

    if(write_size > len) write_size = len;

    while(len > 0)
    {
        HAL_I2C_Mem_Write(&EEPROM_I2C, EEPROM_ADDR_W, addr, I2C_MEMADD_SIZE_8BIT, pbuf, write_size, 100);
        HAL_Delay(5);

        addr += write_size;
        pbuf += write_size;
        len -= write_size;
        write_size = EEPROM_PAGE_SIZE;
        if(len < write_size) write_size = len;
    }
}
```

```c
/**

void USART_Init(void)
{

    HAL_UART_Receive_IT(&USART_HANDLE, &USART_RX_TEMP, 1);
}

/**
  * @brief  串口中断服务函数（处理空闲中断）
  */
void USART1_IRQHandler(void)
{
    HAL_UART_IRQHandler(&USART_HANDLE);

    // 空闲中断 = 一帧数据接收完毕
    if(__HAL_UART_GET_FLAG(&USART_HANDLE, UART_FLAG_IDLE) != RESET)
    {
        __HAL_UART_CLEAR_IDLEFLAG(&USART_HANDLE);
        USART_RX_FLAG = 1; // 标记接收完成
    }
}

/**
  * @brief  单字节接收完成回调
  */
void HAL_UART_RxCpltCallback(UART_HandleTypeDef *huart)

    if(huart->Instance == USART1)
    {
        if(USART_RX_LEN < USART_RX_BUF_SIZE-1)
        {
            USART_RX_BUF[USART_RX_LEN++] = USART_RX_TEMP;
        }
        // 继续接收下一字节
        HAL_UART_Receive_IT(&USART_HANDLE, &USART_RX_TEMP, 1);
    }
}

/**
  * @brief  发送字符串

  * @brief  发送整数
  */

  */

    char buf[32];
    sprintf(buf,"%.*f",dec,num);
    USART_SendString(buf);

/**
  * @brief  指令解析（比赛常用指令）
  */
void USART_ParseCmd(void)
{
    if(!USART_RX_FLAG) return;

    USART_SendString("\r\nRecv: ");
    HAL_UART_Transmit(&USART_HANDLE, USART_RX_BUF, USART_RX_LEN, 100);
    USART_SendString("\r\n");

    // ========== 指令1：LED控制 ==========
    if(strstr((char*)USART_RX_BUF,"LED1_ON") != NULL)
    {
        LED_On(1);
        USART_SendString("OK: LED1 ON\r\n");
    }
    else if(strstr((char*)USART_RX_BUF,"LED1_OFF") != NULL)
    {
        LED_Off(1);
        USART_SendString("OK: LED1 OFF\r\n");
    }
    else if(strstr((char*)USART_RX_BUF,"LED1_TOGGLE") != NULL)
    {
        LED_Toggle(1);
        USART_SendString("OK: LED1 TOGGLE\r\n");
    }

    // ========== 指令2：PWM占空比 ==========
    else if(strstr((char*)USART_RX_BUF,"PWM") != NULL)
    {
        uint8_t duty = atoi((char*)&USART_RX_BUF[3]);
        if(duty>100) duty=100;
        PWM_SetDuty(duty);
        USART_SendString("OK: PWM Duty=");
        USART_SendNum(duty);
        USART_SendString("%\r\n");
    }

    // ========== 指令3：EEPROM写 ==========
    else if(strstr((char*)USART_RX_BUF,"SAVE") != NULL)
    {
        uint8_t val = atoi((char*)&USART_RX_BUF[4]);
        EEPROM_WriteByte(0x00, val);
        USART_SendString("OK: Save ");
        USART_SendNum(val);
        USART_SendString(" to 0x00\r\n");
    }

    // ========== 指令4：EEPROM读 ==========
    else if(strstr((char*)USART_RX_BUF,"READ") != NULL)
    {
        uint8_t val = EEPROM_ReadByte(0x00);
        USART_SendString("OK: Read 0x00 = ");
        USART_SendNum(val);
        USART_SendString("\r\n");
    }

    // 清空接收
    memset(USART_RX_BUF,0,USART_RX_LEN);
    USART_RX_LEN = 0;
    USART_RX_FLAG = 0;
}
```

```c
    // KEY3+ / KEY4-
    if(KEY_GetState(KEY3) == KEY_CLICK && duty < 100) duty += 10;
    if(KEY_GetState(KEY4) == KEY_CLICK && duty > 0)  duty -= 10;
}

/**
 * @brief  ADC模式
 */
static void State_Adc(void)
{
    uint16_t adc = ADC_GetAverageSample(10);
    float vol = adc * 3.3f / 4095.0f;

    LCD_ShowString(0,1,"==ADC MODE==");
    LCD_ShowString(0,2,"Vol:");
    LCD_ShowFloat(5,2,vol,4,2);
    LCD_ShowString(0,3,"STATE:ADC");
}

/**
 * @brief  状态切换（按键KEY1下一个，KEY2上一个）
 */
static void State_Switch(void)
{
    if(KEY_GetState(KEY1) == KEY_CLICK && last_key != 1)
    {
        last_key = 1;
        NowState++;
        if(NowState >= STATE_MAX) NowState = 0;
        LCD_Clear(); // 切换清屏
    }

    if(KEY_GetState(KEY2) == KEY_CLICK && last_key != 2)
    {
        last_key = 2;
        if(NowState == 0) NowState = STATE_MAX-1;
        else NowState--;
        LCD_Clear();
    }

    // 松开复位
    if(KEY_GetState(KEY1)==KEY_RELEASE && KEY_GetState(KEY2)==KEY_RELEASE)
        last_key = 0;
}

/**
 * @brief  状态机总调度（100ms调用一次）
 */
void State_Process(void)
{
    State_Switch(); // 按键切换状态

    // 状态调度
    switch(NowState)
    {
        case STATE_MAIN: State_Main(); break;
        case STATE_LED:  State_Led();  break;
        case STATE_PWM:  State_Pwm();  break;
        case STATE_ADC:  State_Adc();  break;
        default: break;
    }
}
```
